1// crates/docsbookgen/src/main.rs
2use std::{fs, path::Path, process::Command};
3
4fn main() {
5    println!("üìò Running docsbookgen...");
6
7    let mdbook_src = Path::new("docs/mdbook");
8    let docbook_out = Path::new("docsbook");
9
10    // Ensure the docsbook output directory exists
11    if !docbook_out.exists() {
12        fs::create_dir_all(docbook_out).expect("Failed to create docsbook dir");
13    }
14
15    // Step 1: Generate Rust documentation via cargo doc
16    println!("ü¶Ä Generating Rust API docs...");
17    let status = Command::new("cargo")
18        .args(["doc", "--workspace", "--no-deps"])
19        .status()
20        .expect("Failed to run cargo doc");
21    assert!(status.success(), "cargo doc failed");
22
23    // Step 2: Copy cargo doc to docsbook/api
24    let api_doc_path = docbook_out.join("api");
25    if api_doc_path.exists() {
26        fs::remove_dir_all(&api_doc_path).expect("Failed to clear old API docs");
27    }
28    fs::create_dir_all(&api_doc_path).unwrap();
29    fs_extra::dir::copy(
30        "target/doc",
31        &api_doc_path,
32        &fs_extra::dir::CopyOptions::new()
33            .overwrite(true)
34            .copy_inside(true),
35    )
36    .expect("Failed to copy API docs");
37
38    // Step 3: Build mdbook (if exists)
39    if mdbook_src.exists() {
40        println!("üìö Building mdbook...");
41        let status = Command::new("mdbook")
42            .arg("build")
43            .arg(mdbook_src)
44            .status()
45            .expect("Failed to build mdbook");
46        assert!(status.success(), "mdbook build failed");
47
48        // Step 4: Copy mdbook output to docsbook/md
49        let mdbook_output_path = mdbook_src.join("book");
50        let dest = docbook_out.join("md");
51
52        if dest.exists() {
53            fs::remove_dir_all(&dest).expect("Failed to clear old mdbook");
54        }
55        fs::create_dir_all(&dest).unwrap();
56        fs_extra::dir::copy(
57            &mdbook_output_path,
58            &dest,
59            &fs_extra::dir::CopyOptions::new()
60                .overwrite(true)
61                .copy_inside(true),
62        )
63        .expect("Failed to copy mdbook docs");
64    } else {
65        println!("‚ö†Ô∏è  No docs/mdbook/ folder found ‚Äî skipping mdbook build");
66    }
67
68    // Step 5: Copy the custom CSS used by the index page
69    let style_src = Path::new("docsbook/style/tinkerbell.css");
70    let style_dest = docbook_out.join("style");
71    if style_src.exists() {
72        fs::create_dir_all(&style_dest).expect("Failed to create docsbook/style dir");
73        fs::copy(&style_src, style_dest.join("tinkerbell.css")).expect("Failed to copy CSS");
74    }
75
76    // Step 6: Generate a simple index.html linking to API and mdBook docs
77    let index_contents = r#"<!DOCTYPE html>
78<html lang=\"en\">
79  <head>
80    <meta charset=\"utf-8\">
81    <title>Tinkerbell Documentation</title>
82    <link rel=\"stylesheet\" href=\"style/tinkerbell.css\">
83  </head>
84  <body>
85    <h1>Tinkerbell Documentation</h1>
86    <ul>
87      <li><a href=\"api/index.html\">Rust API Docs</a></li>
88      <li><a href=\"md/book/index.html\">mdBook Docs</a></li>
89    </ul>
90  </body>
91</html>
92"#;
93
94    fs::write(docbook_out.join("index.html"), index_contents)
95        .expect("Failed to write docsbook/index.html");
96
97    println!("‚úÖ docsbookgen completed. Docs available in ./docsbook/");
98}
